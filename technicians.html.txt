  import { initializeApp } from "https://www.gstatic.com/firebasejs/11.10.0/firebase-app.js";

    authDomain: "quickfixcrew-717cc.firebaseapp.com",
    projectId: "quickfixcrew-717cc",
    storageBucket: "quickfixcrew-717cc.appspot.com",
    messagingSenderId: "907519125984",
    appId: "1:907519125984:web:20eced6ae597eb454b2385"
  };

  const app = initializeApp(firebaseConfig);
  const db = getFirestore(app);

  const profileContainer = document.getElementById("profile");

  const urlParams = new URLSearchParams(window.location.search);
  const techId = urlParams.get("id");

  if (!techId) {
    profileContainer.innerHTML = "<p>Technician ID not provided.</p>";
  } else {
    const ref = doc(db, "technicians", techId);
    const loadProfile = async () => {
      const docSnap = await getDoc(ref);
      if (!docSnap.exists()) {
        profileContainer.innerHTML = "<p>Technician not found.</p>";
        return;
      }

      const t = docSnap.data();

      profileContainer.innerHTML = `
        <img src="${t.photoURL || 'https://via.placeholder.com/120'}" alt="${t.name}" class="tech-photo" />
        <h2>${t.name}</h2>
        <div class="rating">${"⭐".repeat(t.rating || 4)}</div>
        <p><strong>Skills:</strong> ${t.skills || "N/A"}</p>
        <p><strong>Specialty:</strong> ${t.specialty || "General Repairs"}</p>
        <p><strong>Location:</strong> ${t.location || "N/A"}</p>
        <p><strong>Contact:</strong> ${t.contact || "N/A"}</p>
        <p><strong>Bio:</strong> ${t.bio || "No bio provided."}</p>
      `;
    };

    loadProfile();
  }
</script>
